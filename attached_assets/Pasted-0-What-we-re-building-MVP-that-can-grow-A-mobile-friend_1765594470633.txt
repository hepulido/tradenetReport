2) Supabase: create tables (paste this SQL)

Go to Supabase → SQL Editor → New query and paste:

-- Enable UUIDs
create extension if not exists "pgcrypto";

-- Companies
create table if not exists companies (
  id uuid primary key default gen_random_uuid(),
  name text not null,
  timezone text not null default 'America/New_York',
  created_at timestamptz not null default now()
);

-- Projects
create table if not exists projects (
  id uuid primary key default gen_random_uuid(),
  company_id uuid not null references companies(id) on delete cascade,
  name text not null,
  status text not null default 'active', -- active|paused|closed
  start_date date,
  end_date date,
  created_at timestamptz not null default now()
);

-- Vendors
create table if not exists vendors (
  id uuid primary key default gen_random_uuid(),
  company_id uuid not null references companies(id) on delete cascade,
  name text not null,
  created_at timestamptz not null default now()
);

-- Normalized transactions (expenses + bills + invoices + payments)
create table if not exists transactions (
  id uuid primary key default gen_random_uuid(),
  company_id uuid not null references companies(id) on delete cascade,
  project_id uuid references projects(id) on delete set null,
  vendor_id uuid references vendors(id) on delete set null,

  type text not null, -- expense|bill|invoice|payment
  direction text not null, -- out|in  (out=cost, in=revenue)

  amount numeric(12,2) not null,
  currency text not null default 'USD',
  txn_date date not null,

  category text, -- materials|labor|subcontract|equipment|permit|other
  memo text,

  source text not null default 'seed', -- seed|qbo|email|manual
  source_ref text, -- external id (qbo txn id, email msg id, etc)

  created_at timestamptz not null default now()
);

-- Labor entries (separate so you can support timesheets even without QB)
create table if not exists labor_entries (
  id uuid primary key default gen_random_uuid(),
  company_id uuid not null references companies(id) on delete cascade,
  project_id uuid references projects(id) on delete set null,

  worker_name text,
  role text, -- foreman|laborer|pm|etc
  hours numeric(8,2) not null,
  rate numeric(10,2), -- hourly rate
  labor_date date not null,

  source text not null default 'seed', -- seed|qbo|email|manual
  source_ref text,

  created_at timestamptz not null default now()
);

-- Weekly reports (store summary JSON + PDF URL)
create table if not exists weekly_reports (
  id uuid primary key default gen_random_uuid(),
  company_id uuid not null references companies(id) on delete cascade,
  week_start date not null,
  week_end date not null,

  summary jsonb not null, -- totals + per-project breakdown + alerts
  pdf_url text, -- later: store in Supabase Storage
  created_at timestamptz not null default now()
);

-- Imports log (for file/email/qbo sync jobs)
create table if not exists imports (
  id uuid primary key default gen_random_uuid(),
  company_id uuid not null references companies(id) on delete cascade,
  source text not null, -- email|qbo|manual
  status text not null default 'received', -- received|processed|failed
  details jsonb,
  created_at timestamptz not null default now()
);

-- Helpful indexes
create index if not exists idx_transactions_company_date on transactions(company_id, txn_date);
create index if not exists idx_transactions_project_date on transactions(project_id, txn_date);
create index if not exists idx_labor_company_date on labor_entries(company_id, labor_date);
create index if not exists idx_reports_company_week on weekly_reports(company_id, week_start);


This schema is production-aligned: it supports seed now, QBO/email later with source + source_ref.

3) Backend API (minimum endpoints)

We’ll build these first:

Companies

POST /api/seed/demo → creates demo company + projects + transactions + labor

GET /api/companies → list

Projects

GET /api/companies/:companyId/projects

GET /api/projects/:projectId/summary?weekStart=YYYY-MM-DD

Reports

POST /api/companies/:companyId/reports/weekly (weekStart) → generates summary JSON + stores to weekly_reports

GET /api/companies/:companyId/reports → list reports

GET /api/reports/:reportId → view report
